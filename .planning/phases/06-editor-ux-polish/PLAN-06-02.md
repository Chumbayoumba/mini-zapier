### Task 1: Create graph-validation.ts utility
**File**: `apps/frontend/src/lib/graph-validation.ts` (NEW)

```typescript
import type { Edge, Connection } from '@xyflow/react';

/**
 * Check if adding a connection would create a cycle (DFS).
 * Returns true if adding source→target would create a cycle.
 */
export function wouldCreateCycle(
  edges: Edge[],
  source: string,
  target: string
): boolean {
  if (source === target) return true; // Self-loop

  const visited = new Set<string>();

  function dfs(nodeId: string): boolean {
    if (nodeId === source) return true;
    if (visited.has(nodeId)) return false;
    visited.add(nodeId);

    return edges
      .filter(e => e.source === nodeId)
      .some(e => dfs(e.target));
  }

  return dfs(target);
}

/**
 * Check if an edge between source and target already exists.
 */
export function isDuplicateEdge(
  edges: Edge[],
  source: string,
  target: string
): boolean {
  return edges.some(e => e.source === source && e.target === target);
}

/**
 * Check if a node is a trigger type.
 */
export function isTriggerNode(nodeType: string | undefined): boolean {
  return nodeType === 'triggerNode';
}

/**
 * Validate a connection attempt.
 * Returns { valid: boolean; reason?: string }
 */
export function validateConnection(
  connection: Connection,
  edges: Edge[],
  nodes: { id: string; type?: string }[]
): { valid: boolean; reason?: string } {
  const { source, target } = connection;

  if (!source || !target) {
    return { valid: false, reason: 'Missing source or target' };
  }

  // Rule 1: No self-loops
  if (source === target) {
    return { valid: false, reason: 'Cannot connect a node to itself' };
  }

  // Rule 2: No duplicate edges
  if (isDuplicateEdge(edges, source, target)) {
    return { valid: false, reason: 'Connection already exists' };
  }

  // Rule 3: No cycles
  if (wouldCreateCycle(edges, source, target)) {
    return { valid: false, reason: 'Connection would create a circular reference' };
  }

  // Rule 4: No connecting trigger→trigger
  const sourceNode = nodes.find(n => n.id === source);
  const targetNode = nodes.find(n => n.id === target);
  if (isTriggerNode(sourceNode?.type) && isTriggerNode(targetNode?.type)) {
    return { valid: false, reason: 'Cannot connect two triggers' };
  }

  // Rule 5: No incoming edges to trigger nodes
  if (isTriggerNode(targetNode?.type)) {
    return { valid: false, reason: 'Triggers cannot receive incoming connections' };
  }

  return { valid: true };
}

/**
 * Count trigger nodes in the graph.
 */
export function countTriggerNodes(nodes: { type?: string }[]): number {
  return nodes.filter(n => isTriggerNode(n.type)).length;
}
```

- Pure functions, no React dependencies
- Cycle detection: DFS from target node, checking if it can reach source
- O(V+E) complexity — acceptable for <50 node workflows
- Returns structured result with reason for error messages

### Task 4: Add multi-trigger guard on node drop
**File**: `apps/frontend/src/app/(dashboard)/workflows/[id]/editor/page.tsx`

Enhance `onDrop` to warn when adding a second trigger:

```tsx
import { countTriggerNodes } from '@/lib/graph-validation';

const onDrop = useCallback((event: React.DragEvent) => {
  event.preventDefault();
  const type = event.dataTransfer.getData('application/reactflow-type');
  const label = event.dataTransfer.getData('application/reactflow-label');
  if (!type) return;

  const isTrigger = ['WEBHOOK', 'CRON', 'EMAIL'].includes(type);

  // Multi-trigger guard
  if (isTrigger) {
    const currentTriggers = countTriggerNodes(useEditorStore.getState().nodes);
    if (currentTriggers >= 1) {
      toast.warning('Workflow can only have one trigger. Remove the existing trigger first.');
      return;
    }
  }

  const position = screenToFlowPosition({ x: event.clientX, y: event.clientY });
  const newNode: Node = {
    id: `${isTrigger ? 'trigger' : 'action'}-${Date.now()}`,
    type: isTrigger ? 'triggerNode' : 'actionNode',
    position,
    data: { label, type, config: {}, description: '' },
  };

  addNode(newNode);
}, [screenToFlowPosition, addNode]);
```

export function AnimatedEdge({
  id, sourceX, sourceY, targetX, targetY,
  sourcePosition, targetPosition, style = {}, markerEnd, selected,

        <animateMotion dur="2s" repeatCount="indefinite" path={edgePath} />

    </>
  );
}

'use client';

import { useEditorStore } from '@/stores/editor-store';
import { Button } from '@/components/ui/button';

export function MultiSelectToolbar() {

      </span>

      <Button
        size="sm"
        variant="ghost"
        className="h-7 gap-1 text-xs"

        size="sm"
        variant="ghost"

      >
        <Trash2 className="h-3 w-3" />
        Delete
      </Button>
    </div>
  );
}

## Acceptance Criteria

| # | Criterion | How to Verify |
|---|-----------|---------------|
| AC-1 | Cycle detection: A→B→C, connecting C→A is rejected | Create chain, try closing loop → edge rejected + toast |
| AC-2 | Self-loop: connecting node to itself is rejected | Drag from source handle back to same node → rejected |
| AC-3 | Duplicate edge: same source→target rejected | Connect A→B, try connecting A→B again → rejected |
| AC-4 | Multi-trigger guard: only 1 trigger per workflow | Drop trigger, try dropping second → warning toast |
| AC-5 | Trigger→trigger connection rejected | Try connecting two triggers → rejected |
| AC-6 | Incoming edge to trigger rejected | Try connecting action→trigger target → rejected |
| AC-7 | Connection line shows dashed purple during valid drag | Drag from handle → see dashed indigo line |
| AC-8 | Handles glow on hover | Hover over handle → scale + glow effect |
| AC-9 | Canvas shows dashed outline during drag-over | Drag node from sidebar → canvas gets dashed border |
| AC-10 | Multi-select toolbar shows with ≥2 selected nodes | Shift+click 2 nodes → floating toolbar appears |
| AC-11 | Multi-select delete works | Select 3 nodes → click Delete in toolbar → all removed |
| AC-12 | Selected edges show indigo highlight | Click an edge → thicker indigo stroke |

## Risk Mitigations

| Risk | Mitigation |
|------|------------|
| DFS cycle detection slow on large graphs | Our workflows are <50 nodes; DFS is O(V+E), negligible |
| isValidConnection fires frequently during drag | Pure function, no state updates, no side effects |
| Toast spam on repeated invalid attempts | Toasts auto-dismiss; consider debouncing in future |
| CSS classes conflict with React Flow internals | Use specific class selectors, test in both light/dark modes |
| deleteKeyCode={null} breaks other React Flow delete behavior | Our hook handles all delete logic; React Flow default only handles backspace |
