### Task 2: Create useAutoSave hook
**File**: `apps/frontend/src/hooks/use-auto-save.ts` (NEW)

```typescript
'use client';

import { useEffect, useRef, useCallback } from 'react';
import { useEditorStore } from '@/stores/editor-store';
import { useUpdateWorkflow } from '@/hooks/use-workflows';

const AUTO_SAVE_DELAY = 2000; // 2 seconds debounce

export function useAutoSave(workflowId: string) {
  const updateWorkflow = useUpdateWorkflow();
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const isMountedRef = useRef(true);

  const save = useCallback(async () => {
    const { nodes, edges, isDirty, markSaving, markSaved } = useEditorStore.getState();
    if (!isDirty || nodes.length === 0) return;

    markSaving();
    try {
      await updateWorkflow.mutateAsync({
        id: workflowId,
        definition: { nodes, edges },
      });
      if (isMountedRef.current) {
        markSaved();
      }
    } catch {
      // Save failed — isDirty stays true, will retry on next change
      if (isMountedRef.current) {
        useEditorStore.setState({ isSaving: false });
      }
    }
  }, [workflowId, updateWorkflow]);

  // Subscribe to store changes and debounce save
  useEffect(() => {
    const unsubscribe = useEditorStore.subscribe((state, prevState) => {
      // Only trigger auto-save when nodes or edges change
      if (state.nodes !== prevState.nodes || state.edges !== prevState.edges) {
        if (timerRef.current) clearTimeout(timerRef.current);
        timerRef.current = setTimeout(() => {
          save();
        }, AUTO_SAVE_DELAY);
      }
    });

    return () => {
      unsubscribe();
      if (timerRef.current) clearTimeout(timerRef.current);
      isMountedRef.current = false;
    };
  }, [save]);

  // Expose manual save for Ctrl+S
  const saveNow = useCallback(() => {
    if (timerRef.current) clearTimeout(timerRef.current);
    save();
  }, [save]);

  return { saveNow };
}
```

Key design decisions:
- **Zustand `subscribe`** for reactive change detection (NOT useEffect on state)
- **2-second debounce** prevents excessive API calls during rapid editing
- **Ctrl+S (`saveNow`)** cancels pending debounce and saves immediately
- **`isDirty` check** prevents saving unchanged state
- **Empty nodes guard** prevents saving empty canvas (e.g., during initial load)
- **Cleanup**: clears timer and unsubscribes on unmount
- **Silent toast suppression**: `useUpdateWorkflow` already handles success toast — consider creating a silent variant or suppressing for auto-save (see Task 7)

### Task 3: Create SaveIndicator component
**File**: `apps/frontend/src/components/editor/save-indicator.tsx` (NEW)

```tsx
'use client';

import { useEditorStore } from '@/stores/editor-store';
import { Cloud, CloudOff, Loader2, Check } from 'lucide-react';

export function SaveIndicator() {
  const isDirty = useEditorStore(s => s.isDirty);
  const isSaving = useEditorStore(s => s.isSaving);
  const lastSavedAt = useEditorStore(s => s.lastSavedAt);

  if (isSaving) {
    return (
      <div className="flex items-center gap-1.5 text-xs text-muted-foreground">
        <Loader2 className="h-3 w-3 animate-spin" />
        <span>Saving...</span>
      </div>
    );
  }

  if (isDirty) {
    return (
      <div className="flex items-center gap-1.5 text-xs text-amber-500">
        <CloudOff className="h-3 w-3" />
        <span>Unsaved changes</span>
      </div>
    );
  }

  if (lastSavedAt) {
    const time = lastSavedAt.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
    return (
      <div className="flex items-center gap-1.5 text-xs text-green-600 dark:text-green-400">
        <Check className="h-3 w-3" />
        <span>Saved at {time}</span>
      </div>
    );
  }

  return (
    <div className="flex items-center gap-1.5 text-xs text-muted-foreground">
      <Cloud className="h-3 w-3" />
      <span>Draft</span>
    </div>
  );
}
```

States in priority order:
1. **Saving** → spinner + "Saving..."
2. **Dirty** → amber warning + "Unsaved changes"
3. **Saved** → green check + "Saved at HH:MM"
4. **Default** → cloud + "Draft" (initial state before any save)

### Task 4: Extract toolbar to EditorToolbar component
**File**: `apps/frontend/src/components/editor/editor-toolbar.tsx` (NEW)

```tsx
'use client';

import { useRouter } from 'next/navigation';
import { useEditorStore } from '@/stores/editor-store';
import { useActivateWorkflow } from '@/hooks/use-workflows';
import { useViewport } from '@xyflow/react';
import { useStoreWithEqualityFn } from 'zustand/traditional';
import { Button } from '@/components/ui/button';
import { SaveIndicator } from '@/components/editor/save-indicator';
import {
  ArrowLeft, Maximize, Play, Undo2, Redo2,
  Upload, ZoomIn,
} from 'lucide-react';
import { toast } from 'sonner';

interface EditorToolbarProps {
  workflowId: string;
  workflowName?: string;
  workflowStatus?: string;
  onFitView: () => void;
  onRun: () => void;
  onSaveNow: () => void;
}

export function EditorToolbar({
  workflowId, workflowName, workflowStatus,
  onFitView, onRun, onSaveNow,
}: EditorToolbarProps) {
  const router = useRouter();
  const activateWorkflow = useActivateWorkflow();
  const { zoom } = useViewport();

  const canUndo = useStoreWithEqualityFn(
    useEditorStore.temporal,
    (s) => s.pastStates.length > 0
  );
  const canRedo = useStoreWithEqualityFn(
    useEditorStore.temporal,
    (s) => s.futureStates.length > 0
  );
  const isDirty = useEditorStore(s => s.isDirty);
  const nodeCount = useEditorStore(s => s.nodes.length);

  const handlePublish = async () => {
    if (isDirty) {
      onSaveNow(); // Save first
    }

    if (nodeCount === 0) {
      toast.error('Add at least one node before publishing');
      return;
    }

    const confirmed = window.confirm(
      'Publish this workflow? It will become active and start processing triggers.'
    );
    if (!confirmed) return;

    try {
      await activateWorkflow.mutateAsync(workflowId);
      toast.success('Workflow published and active');
    } catch {
      toast.error('Failed to publish workflow');
    }
  };

  const isActive = workflowStatus === 'ACTIVE';

  return (
    <div className="flex items-center gap-2 border-b px-3 py-2 bg-card">
      {/* Navigation */}
      <Button size="sm" variant="ghost" className="gap-1.5 text-muted-foreground mr-1" onClick={() => router.push('/workflows')}>
        <ArrowLeft className="h-3.5 w-3.5" />
        <span className="hidden sm:inline text-xs">Back</span>
      </Button>

      <div className="w-px h-5 bg-border" />

      {/* Undo/Redo */}
      <Button size="sm" variant="ghost" onClick={() => useEditorStore.temporal.getState().undo()} disabled={!canUndo} title="Undo (Ctrl+Z)">
        <Undo2 className="h-3.5 w-3.5" />
      </Button>
      <Button size="sm" variant="ghost" onClick={() => useEditorStore.temporal.getState().redo()} disabled={!canRedo} title="Redo (Ctrl+Y)">
        <Redo2 className="h-3.5 w-3.5" />
      </Button>

      <div className="w-px h-5 bg-border" />

      {/* Workflow name + save indicator */}
      <h2 className="font-semibold text-sm truncate px-1">
        {workflowName || 'Workflow Editor'}
      </h2>
      <SaveIndicator />

      <div className="flex-1" />

      {/* Zoom indicator */}
      <div className="flex items-center gap-1 text-xs text-muted-foreground" title="Zoom level">
        <ZoomIn className="h-3 w-3" />
        <span>{Math.round(zoom * 100)}%</span>
      </div>

      <div className="w-px h-5 bg-border" />

      {/* Fit view */}
      <Button size="sm" variant="ghost" onClick={onFitView} title="Fit to view">
        <Maximize className="h-3.5 w-3.5" />
      </Button>

      <div className="w-px h-5 bg-border" />

      {/* Publish */}
      <Button
        size="sm"
        variant={isActive ? 'secondary' : 'outline'}
        className="gap-1.5"
        onClick={handlePublish}
        disabled={activateWorkflow.isPending || isActive}
        title={isActive ? 'Workflow is already active' : 'Publish workflow'}
      >
        <Upload className="h-3.5 w-3.5" />
        {isActive ? 'Published' : 'Publish'}
      </Button>

      {/* Run */}
      <Button size="sm" className="gap-1.5 shadow-sm shadow-primary/20" onClick={onRun}>
        <Play className="h-3.5 w-3.5" />
        Run
      </Button>
    </div>
  );
}
```

Toolbar layout (left to right):
1. Back button
2. Undo / Redo buttons
3. Workflow name + SaveIndicator
4. (flex spacer)
5. Zoom indicator (e.g., "125%")
6. Fit view button
7. Publish button
8. Run button

### Task 7: Create silent update mutation for auto-save
**File**: `apps/frontend/src/hooks/use-workflows.ts`

Add a silent variant that doesn't show "Workflow updated" toast on every auto-save:

```typescript
export function useUpdateWorkflowSilent() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: async ({ id, ...data }: { id: string; [key: string]: any }) => {
      const res = await api.patch(`/workflows/${id}`, data);
      return res.data.data || res.data;
    },
    onSuccess: (_, variables) => {
      queryClient.invalidateQueries({ queryKey: ['workflow', variables.id] });
      // No toast — auto-save should be silent
    },
  });
}
```

### Task 9: Add leave guard (beforeunload)
**File**: `apps/frontend/src/app/(dashboard)/workflows/[id]/editor/page.tsx`

Add `beforeunload` event listener to warn about unsaved changes:

```tsx
// Inside EditorCanvas component:
useEffect(() => {
  const handler = (e: BeforeUnloadEvent) => {
    const { isDirty } = useEditorStore.getState();
    if (isDirty) {
      e.preventDefault();
      // Modern browsers show generic message regardless of returnValue
      return '';
    }
  };

  window.addEventListener('beforeunload', handler);
  return () => window.removeEventListener('beforeunload', handler);
}, []);
```

- Only triggers when `isDirty === true`
- Modern browsers show their own generic "Leave site?" dialog
- Cleared on unmount

## Risk Mitigations

| Risk | Mitigation |
|------|------------|
| Auto-save conflicts with manual Ctrl+S | `saveNow()` clears pending debounce timer first |
| Toast spam from auto-save | Use `useUpdateWorkflowSilent` — no toast on success |
| Auto-save during initial load | Guard: `if (!isDirty || nodes.length === 0) return` |
| `useViewport()` outside ReactFlowProvider | `EditorToolbar` rendered inside `EditorCanvas` which is inside `ReactFlowProvider` ✓ |
| beforeunload not working in Next.js | Uses raw `window.addEventListener`, works in all browsers |
| Panel slide animation jank | Tailwind `animate-in` uses CSS transforms, GPU-accelerated |
| Auto-save fails silently | On error: `isSaving` reset to false, `isDirty` stays true → retries on next change |
